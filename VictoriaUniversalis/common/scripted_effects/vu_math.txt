vu_pow = {
	set_variable = {
		which = exponent
		which = $EXPONENT$
	}
	set_variable = {
		which = base
		which = $BASE$
	}
	set_variable = {
		which = pow
		which = $BASE$
	}
	if = {
		NOT = {
			check_variable = {
				which = exponent
				value = 0
			}
		}
		multiply_variable = {
			which = exponent
			value = -1
		}
	}
	while = {
		limit = {
			check_variable = {
				which = exponent
				value = 2
			}
		}
		change_variable = {
			which = exponent
			value = -1
		}
		multiply_variable = {
			which = pow
			which = base
		}
	}
	set_variable = {
		which = exponent
		which = $EXPONENT$
	}
	if = {
		NOT = {
			check_variable = {
				which = exponent
				value = 0
			}
		}
		set_variable = {
			which = pow_output
			value = 1
		}
		divide_variable = {
			which = pow_output
			which = pow
		}
	}
	else = {
		set_variable = {
			which = pow_output
			which = pow
		}
	}
	
}
